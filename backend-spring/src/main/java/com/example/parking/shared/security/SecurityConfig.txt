package com.example.parking.shared.security;

import java.io.Console;

import org.springframework.context.annotation.Bean;
import org.springframework.context.annotation.Configuration;
import org.springframework.security.config.annotation.web.builders.HttpSecurity;
import org.springframework.security.config.annotation.web.configurers.AbstractHttpConfigurer;
import org.springframework.security.crypto.bcrypt.BCryptPasswordEncoder;
import org.springframework.security.crypto.password.PasswordEncoder;
import org.springframework.security.web.SecurityFilterChain;

@Configuration// cái annotation này là j , nó từ đâu ra , tại sao phải dùng nó , ko có nó thì sao , trường hợp mới dùng nó
// cái này là 1 annotation của spring và nó nói với spring rằng đây ko phải là 1 class nghiệp vụ mà là 1 class config và nhờ spring quản lý
// thế tại sao phải dùng nó?
//  - Vì trong class này có các bean và Spring cần biết đây là 1 file cofig để đọc các bean bên trong
//  - Nếu ko có thì Spring ko quét class này như 1 file cấu hình mà 
public class SecurityConfig {

    @Bean
    public SecurityFilterChain securityFilterChain(HttpSecurity http/* HttpSecurity là 1 object do Spring Security đưa vào , dùng để cấu hình bảo mật HTTP */) throws Exception {// cái này là j , nó từ đâu ra , tại sao phải dùng nó , ko có nó thì sao , trường hợp mới dùng nó
        // Câu hỏi đặt ra SecurityFilterChain là j ?
        //  - Là chuỗi bộ lọc bảo mật của Spring Security
        //  CỤ thể là nó quyết định :
        //      + Request nào được đi qua
        //      + Request nào bị chặn
        //      + API nào cần login 
        //      + Có bật CSRF ko ?
        //      + Có dùng HTTP basic ko ?
        //      + Có dùng JWT Filter ko ?
        //      ....
        // Nói 1 cách dễ hiểu nó là 1 bảo vệ đứng gác để ko phải cứ reqquest nào cx đc đi vào backend và phải đi qua nó trước
        // Nó đến từ : Spring security!
        
        // Nếu ko có nó thì khi sử dụng denpency Spring security thfi sẽ dùng mặc định sẽ có các hiện tượng:
//          - Gọi API bị 401 Unauthorized
//          - Trình duyệt hiện form login mặc định
//          - Console in ra generated password
//          - API register/login có thể bị chặn
        return http
        // http là 1 object HttpSecurity như 1 cái bảng điều khiển để cấu hình bảo mật
                .csrf(AbstractHttpConfigurer::disable)// CSRF là cơ chế chống tấn công giả mạo request trong web dùng session/cookie.
                .formLogin(AbstractHttpConfigurer::disable)//Dòng này tắt form login mặc định của Spring Security vì Mình không muốn Spring tự hiện trang HTML login.
                .httpBasic(AbstractHttpConfigurer::disable)// Dòng này tắt HTTP Basic Authentication.Tại sao phải tắt?
                                                        //     Vì mình không muốn dùng kiểu gửi username/password trong Basic Auth.
                .authorizeHttpRequests(auth -> auth     // Đây là phần cấu hình quyền truy cập cho request. Đây là nơi mình quy định URL nào được gọi tự do, URL nào cần đăng nhập.
                        .requestMatchers("/api/auth/**").permitAll()// Tất cả API bắt đầu bằng /api/auth/ được phép gọi mà không cần đăng nhập.
                        .anyRequest()
                        .permitAll()// 2 hàm này là Tất cả request còn lại cũng được phép gọi tự do.
                )
                .build();// Sau khi cấu hình xong http, mình gọi .build() để tạo ra object:
    }

    @Bean 
    // Vì đã có Annotation là Configuration nên Springs sẽ tạo ra object của các bean bên trong nó là PasswwordEncoder ở dưới đây
    public PasswordEncoder passwordEncoder() {// cái này là j , nó từ đâu ra , tại sao phải dùng nó , ko có nó thì sao , trường hợp mới dùng nó
        return new BCryptPasswordEncoder();
    }
}

// Tổng thể đây là file cấu hình security cho cả spring boot
//      => thẳng ra đây là nới mà người lập trình nói với Spring Security rằng:
//          API nào được phép gọi tự do , API nào cần đăng nhập , có dùng form login ko có bật CSRF ko , dùng loại mã hóa password nào
